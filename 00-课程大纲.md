**必须掌握的 C 特性**：
```c
// 指针运算 — 内核代码天天用
int arr[10];
int *p = arr;
*(p + 5) = 42;  // 等价于 arr[5]

// 函数指针 — 驱动 file_operations 结构体
struct file_operations {
    ssize_t (*read)(struct file *, char __user *, size_t, loff_t *);
    ssize_t (*write)(struct file *, const char __user *, size_t, loff_t *);
};

// container_of — 内核核心宏
#define container_of(ptr, type, member) ({          \
    const typeof(((type *)0)->member) *__mptr = (ptr); \
    (type *)((char *)__mptr - offsetof(type, member)); })

// 位运算 — 寄存器操作
#define BIT(n)  (1UL << (n))
#define REG_SET(reg, bits)   writel(readl(reg) | (bits), (reg))
#define REG_CLR(reg, bits)   writel(readl(reg) & ~(bits), (reg))

// volatile — 防止编译器优化 MMIO 访问
volatile uint32_t *reg = (volatile uint32_t *)0x1c00000;

// list_head — 内核双向链表
struct list_head {
    struct list_head *next, *prev;
};
```

**周末项目：内存池分配器**

```c
/*
 * 实现一个固定大小的内存池分配器：
 * - pool_create(size_t block_size, int count) - 创建池
 * - pool_alloc(pool) - 分配一个块
 * - pool_free(pool, ptr) - 释放一个块
 * - pool_destroy(pool) - 销毁池
 * - 使用位图管理空闲块
 * - O(1) 分配和释放
 */
```
