# 外设驱动 — I2C / SPI / UART / MMC / PCIe

## 概述

外设驱动是内核驱动开发的主要内容。高通 SoC 通过多种外设总线连接各类硬件 — sensor（I2C）、显示和射频（SPI）、蓝牙和 GPS（UART）、存储（MMC/eMMC/UFS）、WiFi（PCIe/SDIO）。

> 高通工作场景：你需要为 Qualcomm 平台开发和调试各类外设驱动，确保硬件能和系统正常通信。

## I2C 驱动

I2C 是传感器类设备最常用的总线（2 线：SCL + SDA）。

### I2C 子系统架构

```
用户空间 → /dev/i2c-N (i2c-dev)
                ↓
I2C 设备驱动 (client driver)   ← 你的驱动
  ├── i2c_driver.match_table  → 匹配 compatible
  ├── i2c_driver.probe        → 初始化
  └── i2c_client              → 设备实例
    
I2C 控制器驱动 (adapter driver) ← 高通的适配器
  ├── i2c_algorithm.master_xfer → 传输函数
  └── i2c_adapter               → 总线控制器

硬件 I2C 控制器
```

### 编写 I2C 设备驱动

```c
#include <linux/i2c.h>
#include <linux/module.h>
#include <linux/regmap.h>

struct sensor_data {
    struct i2c_client *client;
    struct regmap *regmap;
    int irq;
};

static const struct regmap_config sensor_regmap = {
    .reg_bits = 8,
    .val_bits = 8,
    .max_register = 0xFF,
};

static int sensor_probe(struct i2c_client *client)
{
    struct sensor_data *data;
    int ret;
    
    // 分配设备结构
    data = devm_kzalloc(&client->dev, sizeof(*data), GFP_KERNEL);
    if (!data)
        return -ENOMEM;
    
    data->client = client;
    i2c_set_clientdata(client, data);
    
    // 初始化 regmap（封装 I2C 读写）
    data->regmap = devm_regmap_init_i2c(client, &sensor_regmap);
    
    // 读取设备 ID
    u32 chip_id;
    regmap_read(data->regmap, 0x0F, &chip_id);
    dev_info(&client->dev, "Sensor chip ID: 0x%x\n", chip_id);
    
    // 请求中断
    data->irq = client->irq;
    ret = devm_request_threaded_irq(&client->dev, data->irq, NULL,
                                     sensor_irq_thread,
                                     IRQF_TRIGGER_RISING | IRQF_ONESHOT,
                                     "sensor", data);
    return 0;
}

static irqreturn_t sensor_irq_thread(int irq, void *dev_id)
{
    struct sensor_data *data = dev_id;
    // 读取传感器数据
    // ...
    return IRQ_HANDLED;
}

static const struct i2c_device_id sensor_id_table[] = {
    { "my_sensor", 0 },
    { }
};
MODULE_DEVICE_TABLE(i2c, sensor_id_table);

static const struct of_device_id sensor_of_match[] = {
    { .compatible = "myvendor,my-sensor" },
    { }
};
MODULE_DEVICE_TABLE(of, sensor_of_match);

static struct i2c_driver sensor_driver = {
    .driver = {
        .name = "my_sensor",
        .of_match_table = sensor_of_match,
    },
    .probe_new  = sensor_probe,
    .id_table   = sensor_id_table,
};
module_i2c_driver(sensor_driver);
```

## SPI 驱动

SPI 是高速全双工总线（4 线：MOSI/MISO/SCLK/CS），用于显示、RF 等场景。

```c
#include <linux/spi/spi.h>

static int spi_device_probe(struct spi_device *spi)
{
    // SPI 配置
    spi->mode = SPI_MODE_0;      // CPOL=0, CPHA=0
    spi->max_speed_hz = 10000000; // 10MHz
    spi_setup(spi);
    
    // SPI 传输
    struct spi_transfer t = {
        .tx_buf = tx_buffer,
        .rx_buf = rx_buffer,
        .len = 32,
    };
    
    struct spi_message m;
    spi_message_init(&m);
    spi_message_add_tail(&t, &m);
    spi_sync(spi, &m);  // 同步传输
    
    return 0;
}

static const struct of_device_id spi_of_match[] = {
    { .compatible = "myvendor,my-spi-device" },
    { }
};
MODULE_DEVICE_TABLE(of, spi_of_match);

static struct spi_driver spi_driver = {
    .driver = {
        .name = "my_spi_device",
        .of_match_table = spi_of_match,
    },
    .probe = spi_device_probe,
};
module_spi_driver(spi_driver);
```

## UART 驱动

UART 用于蓝牙、GPS、外接模块等串行通信。

```c
// 内核中 UART 驱动使用 serdev 框架（不再是 tty 层）
#include <linux/serdev.h>

static int uart_device_probe(struct serdev_device *serdev)
{
    // 设置波特率
    serdev_device_set_baudrate(serdev, 115200);
    serdev_device_set_flow_control(serdev, false);
    
    // 注册接收回调
    serdev_device_set_client_ops(serdev, &uart_ops);
    
    // 打开设备
    return serdev_device_open(serdev);
}

static const struct of_device_id uart_of_match[] = {
    { .compatible = "myvendor,my-uart-device" },
    { }
};

static struct serdev_device_driver uart_driver = {
    .driver = {
        .name = "my_uart",
        .of_match_table = uart_of_match,
    },
    .probe = uart_device_probe,
};
```

    status = "okay";

## PCIe 驱动

高通平台使用 PCIe 连接 WiFi、NVMe SSD 等高速外设：

```dts
&pcie0 {
    status = "okay";
    
    num-lanes = <1>;         // x1 lane
    perst-gpio = <&tlmm 35 GPIO_ACTIVE_LOW>;
    wake-gpio = <&tlmm 37 GPIO_ACTIVE_HIGH>;
    
    vddpe-supply = <&pm8150_l8>;
};

// PCIe 设备驱动
static int wifi_pci_probe(struct pci_dev *pdev,
                           const struct pci_device_id *id)
{
    // 启用设备
    pci_enable_device(pdev);
    pci_set_master(pdev);
    
    // 分配 DMA 缓冲区
    dma_set_mask(&pdev->dev, DMA_BIT_MASK(64));
    
    // 读取 BAR 地址
    resource_size_t bar_start = pci_resource_start(pdev, 0);
    
    return 0;
}

static const struct pci_device_id wifi_pci_ids[] = {
    { PCI_DEVICE(0x17cb, 0x1101) },  // Qualcomm QCA6390
    { }
};
MODULE_DEVICE_TABLE(pci, wifi_pci_ids);

static struct pci_driver wifi_driver = {
    .name = "qcom_wifi",
    .id_table = wifi_pci_ids,
    .probe = wifi_pci_probe,
    .remove = wifi_pci_remove,
};
```

## 面试考点

1. **I2C 和 SPI 的区别？** I2C 2线多设备（地址寻址），SPI 4线单设备（片选），SPI 更快
2. **regmap 的作用？** 封装 I2C/SPI/MMIO 的寄存器读写，提供缓存和同步机制
3. **高通 GENI 是什么？** Generic Interface，高通的可配置外设接口 IP，可配置为 I2C/SPI/UART
4. **为什么 UFS 比 eMMC 快？** UFS 全双工 + 命令队列，eMMC 半双工 + 单命令
5. **DMA 在驱动中的使用场景？** 大块数据传输（如 camera、audio、存储）避免 CPU 逐字拷贝
