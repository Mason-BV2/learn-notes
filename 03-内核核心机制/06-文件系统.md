# 文件系统 / VFS

## 概述

VFS（Virtual File System）是 Linux 内核的文件系统抽象层。所有文件系统（ext4、btrfs、proc、sysfs、tmpfs）都通过 VFS 接口集成到内核中。

> 高通场景：理解和调试文件系统问题（存储性能、分区表、initramfs、SELinux 策略）、debugfs 调试接口。

## VFS 架构

```
系统调用 (open/read/write/close)
    ↓
VFS 层 (sys_open / vfs_read / vfs_write)
    ↓
具体文件系统 (ext4 / btrfs / proc / sysfs)
    ↓
块设备层 (block I/O / io scheduler)
    ↓
设备驱动 (mmc / ufs / nvme)
```

### VFS 四大对象

```c
// 1. super_block — 已挂载文件系统的实例
struct super_block {
    struct list_head    s_list;         // 所有 super_block 链表
    dev_t               s_dev;          // 设备标识
    unsigned long       s_blocksize;    // 块大小
    struct file_system_type *s_type;    // 文件系统类型
    struct super_operations *s_op;      // 超级块操作函数
    // ...
};

// 2. inode — 文件/目录的元数据（唯一标识一个文件）
struct inode {
    umode_t             i_mode;         // 文件类型和权限
    uid_t               i_uid;          // 用户ID
    gid_t               i_gid;          // 组ID
    loff_t              i_size;         // 文件大小
    struct inode_operations *i_op;      // inode 操作
    struct file_operations  *i_fop;     // 默认的文件操作（!）
    // ...
};

// 3. dentry — 目录项（路径名到 inode 的映射）
struct dentry {
    struct qstr         d_name;         // 文件名
    struct inode        *d_inode;       // 指向的 inode
    struct dentry       *d_parent;      // 父目录
    struct list_head    d_subdirs;      // 子目录链表
    struct dentry_operations *d_op;     // dentry 操作
    // ...
};

// 4. file — 打开的文件实例
struct file {
    struct path         f_path;         // 文件路径 (dentry + mount)
    struct inode        *f_inode;       // 指向的 inode
    const struct file_operations *f_op; // 文件操作函数
    loff_t              f_pos;          // 当前读写位置
    void                *private_data;  // 驱动私有数据（!）
    // ...
};
```

## 驱动中的文件操作

### 1. debugfs — 调试文件系统

debugfs 是驱动开发者最常用的文件接口，不需要注册设备号：

```c
#include <linux/debugfs.h>

static struct dentry *qcom_debug_dir;
static struct dentry *qcom_debug_file;

static u32 debug_reg_val;

static int debug_open(struct inode *inode, struct file *file)
{
    file->private_data = inode->i_private;
    return 0;
}

static ssize_t debug_read(struct file *file, char __user *buf,
                           size_t len, loff_t *off)
{
    char kbuf[32];
    int n = snprintf(kbuf, sizeof(kbuf), "0x%x\n", debug_reg_val);
    return simple_read_from_buffer(buf, len, off, kbuf, n);
}

static ssize_t debug_write(struct file *file, const char __user *buf,
                            size_t len, loff_t *off)
{
    char kbuf[32];
    if (len > sizeof(kbuf) - 1)
        return -EINVAL;
    if (copy_from_user(kbuf, buf, len))
        return -EFAULT;
    kbuf[len] = '\0';
    if (kstrtou32(kbuf, 0, &debug_reg_val))
        return -EINVAL;
    return len;
}

static const struct file_operations debug_fops = {
    .owner = THIS_MODULE,
    .open = debug_open,
    .read = debug_read,
    .write = debug_write,
};

// 在 probe 中创建 debugfs 文件
static int qcom_probe(struct platform_device *pdev)
{
    qcom_debug_dir = debugfs_create_dir("qcom_driver", NULL);
    if (!qcom_debug_dir)
        return -ENOENT;
    
    qcom_debug_file = debugfs_create_file("reg_val", 0644,
                                           qcom_debug_dir, NULL, &debug_fops);
    return 0;
}

// 使用：cat /sys/kernel/debug/qcom_driver/reg_val
// 使用：echo 0xFF > /sys/kernel/debug/qcom_driver/reg_val
```

### 2. debugfs 简化辅助函数

```c
// 不用自己实现 fops，debugfs 提供了辅助函数

// u32 变量的读写
debugfs_create_u32("reg_val", 0644, qcom_debug_dir, &debug_reg_val);

// bool 变量
debugfs_create_bool("enable", 0644, qcom_debug_dir, &debug_enable);

// 寄存器 dump（二进制块）
debugfs_create_blob("regs", 0644, qcom_debug_dir, &blob);

// x86/ARM 寄存器位宽版本
debugfs_create_x32("status", 0644, qcom_debug_dir, &status_reg);
debugfs_create_x64("counter", 0644, qcom_debug_dir, &counter);
```

### 3. sysfs — 内核对象属性

sysfs 用于导出内核对象的属性，以目录结构反映设备拓扑：

```c
#include <linux/sysfs.h>

// 设备属性
static ssize_t foo_show(struct device *dev,
                         struct device_attribute *attr, char *buf)
{
    return sprintf(buf, "%d\n", some_value);
}

static ssize_t foo_store(struct device *dev,
                          struct device_attribute *attr,
                          const char *buf, size_t count)
{
    sscanf(buf, "%d", &some_value);
    return count;
}

static DEVICE_ATTR_RW(foo);  // 创建 foo_show / foo_store

// 在 probe 中创建 sysfs 文件
static int qcom_probe(struct platform_device *pdev)
{
    device_create_file(&pdev->dev, &dev_attr_foo);
    // 使用：cat /sys/devices/platform/.../foo
}

// 或使用 attribute group 批量创建
static struct attribute *qcom_attrs[] = {
    &dev_attr_foo.attr,
    &dev_attr_bar.attr,
    NULL,
};

static const struct attribute_group qcom_attr_group = {
    .attrs = qcom_attrs,
};

// probe 中：
devm_device_add_group(&pdev->dev, &qcom_attr_group);
```

## 文件系统创建 — 虚拟文件系统

```c
// 最简单的方式：使用 libfs 创建内存文件系统

#include <linux/libfs.h>

static struct dentry *qcom_fs_mount(struct file_system_type *fs_type,
                                     int flags, const char *dev_name,
                                     void *data)
{
    return mount_pseudo(fs_type, "qcom_fs:", NULL, NULL, 0);
}

static struct file_system_type qcom_fs_type = {
    .name = "qcom_fs",
    .mount = qcom_fs_mount,
    .kill_sb = kill_litter_super,
};

// 注册
register_filesystem(&qcom_fs_type);
```

## proc 文件系统

```c
#include <linux/proc_fs.h>

// proc_create 简化版（推荐新内核使用 proc_ops）
static const struct proc_ops qcom_proc_fops = {
    .proc_open = qcom_proc_open,
    .proc_read = seq_read,
    .proc_write = qcom_proc_write,
    .proc_release = single_release,
};

// 使用 seq_file 实现顺序读取
static int qcom_proc_show(struct seq_file *m, void *v)
{
    seq_printf(m, "Driver: qcom_i2c\n");
    seq_printf(m, "IRQ: %d\n", irq_num);
    seq_printf(m, "Clock: %lu Hz\n", clk_rate);
    return 0;
}

static int qcom_proc_open(struct inode *inode, struct file *file)
{
    return single_open(file, qcom_proc_show, NULL);
}

// 创建 /proc/qcom_driver 文件
proc_create("qcom_driver", 0644, NULL, &qcom_proc_fops);
```

## 实操练习

### 1. debugfs 调试接口

```bash
# 挂载 debugfs
mount -t debugfs none /sys/kernel/debug

# 查看高通驱动的 debugfs 文件
ls /sys/kernel/debug/
cat /sys/kernel/debug/qcom_i2c/regs

# 查看文件系统信息
cat /proc/filesystems    # 所有注册的文件系统
cat /proc/mounts         # 已挂载的文件系统
stat -f /               # 文件系统信息
```

### 2. 查看 VFS 对象

```bash
# 查看 inode 信息
stat /path/to/file

# 查看 dentry 缓存
cat /proc/slabinfo | grep dentry

# 查看打开的文件
lsof | head -20
cat /proc/sys/fs/file-nr  # 打开的文件数统计
```

## 面试考点

1. **VFS 的四大对象是什么？** super_block / inode / dentry / file
2. **debugfs 和 sysfs 的区别？** debugfs 是调试接口（可读写任意值），sysfs 是设备属性（1个文件1个值）
3. **inode 和 dentry 的区别？** inode 代表文件的元数据（唯一），dentry 代表路径名（可以有很多个硬链接指向同一个 inode）
4. **page cache 的作用？** 缓存文件数据减少磁盘 I/O，读文件时优先从 page cache 读取
5. **内核中如何创建一个虚拟文件系统？** 使用 register_filesystem + mount_pseudo 或 mount_nodev
