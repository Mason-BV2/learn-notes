# 字符设备驱动

## 概述

字符设备（char device）是 Linux 中最基本的设备类型，以字节流方式读写。几乎所有非存储类硬件（传感器、GPS、音频、相机）在内核中都以字符设备方式暴露。

> 高通工作场景：Qualcomm 平台的 sensor、audio、camera 等外设都通过字符设备驱动与用户空间交互。

## 核心数据结构

### 1. file_operations — 驱动与用户态的接口

```c
#include <linux/fs.h>

struct file_operations qcom_fops = {
    .owner   = THIS_MODULE,
    .open    = qcom_open,
    .release = qcom_release,
    .read    = qcom_read,
    .write   = qcom_write,
    .unlocked_ioctl = qcom_ioctl,  // 控制命令
    .llseek  = qcom_llseek,
    .mmap    = qcom_mmap,          // 高性能内存映射
};
```

### 2. 设备号 — 主设备号 + 次设备号

```c
dev_t devno;
int major = 241;  // 主设备号（可动态分配）

// 动态分配设备号（推荐方式）
alloc_chrdev_region(&devno, 0, 1, "qcom_device");
major = MAJOR(devno);

// 查看已分配的设备号
// cat /proc/devices
```

### 3. cdev — 字符设备的内核表示

```c
struct cdev qcom_cdev;

cdev_init(&qcom_cdev, &qcom_fops);
qcom_cdev.owner = THIS_MODULE;
cdev_add(&qcom_cdev, devno, 1);
```

## 完整驱动框架

### 最小字符设备驱动

```c
#include <linux/module.h>
#include <linux/kernel.h>
#include <linux/fs.h>
#include <linux/cdev.h>
#include <linux/slab.h>
#include <linux/uaccess.h>

#define DEVICE_NAME "qcom_char"
#define BUF_SIZE    4096

struct qcom_device {
    struct cdev cdev;
    char buffer[BUF_SIZE];
    size_t len;
} *qcom_dev;

static int qcom_open(struct inode *inode, struct file *filp)
{
    filp->private_data = container_of(inode->i_cdev,
                                      struct qcom_device, cdev);
    pr_info("qcom_char: opened\n");
    return 0;
}

static int qcom_release(struct inode *inode, struct file *filp)
{
    pr_info("qcom_char: released\n");
    return 0;
}

static ssize_t qcom_read(struct file *filp, char __user *buf,
                          size_t len, loff_t *off)
{
    struct qcom_device *dev = filp->private_data;
    
    if (*off >= dev->len)
        return 0;
    if (*off + len > dev->len)
        len = dev->len - *off;
    
    if (copy_to_user(buf, dev->buffer + *off, len))
        return -EFAULT;
    
    *off += len;
    return len;
}

static ssize_t qcom_write(struct file *filp, const char __user *buf,
                           size_t len, loff_t *off)
{
    struct qcom_device *dev = filp->private_data;
    
    if (len > BUF_SIZE)
        len = BUF_SIZE;
    
    if (copy_from_user(dev->buffer, buf, len))
        return -EFAULT;
    
    dev->len = len;
    pr_info("qcom_char: wrote %zu bytes\n", len);
    return len;
}

static long qcom_ioctl(struct file *filp, unsigned int cmd,
                        unsigned long arg)
{
    struct qcom_device *dev = filp->private_data;
    
    switch (cmd) {
    case 0:  // 清空缓冲区
        memset(dev->buffer, 0, BUF_SIZE);
        dev->len = 0;
        pr_info("qcom_char: buffer cleared\n");
        break;
    default:
        return -EINVAL;
    }
    return 0;
}

static struct file_operations qcom_fops = {
    .owner          = THIS_MODULE,
    .open           = qcom_open,
    .release        = qcom_release,
    .read           = qcom_read,
    .write          = qcom_write,
    .unlocked_ioctl = qcom_ioctl,
};

static dev_t devno;

static int __init qcom_init(void)
{
    // 1. 分配设备号
    alloc_chrdev_region(&devno, 0, 1, DEVICE_NAME);
    
    // 2. 分配设备结构体
    qcom_dev = kzalloc(sizeof(*qcom_dev), GFP_KERNEL);
    if (!qcom_dev) {
        unregister_chrdev_region(devno, 1);
        return -ENOMEM;
    }
    
    // 3. 初始化并添加 cdev
    cdev_init(&qcom_dev->cdev, &qcom_fops);
    qcom_dev->cdev.owner = THIS_MODULE;
    cdev_add(&qcom_dev->cdev, devno, 1);
    
    pr_info("qcom_char: loaded, major=%d\n", MAJOR(devno));
    return 0;
}

static void __exit qcom_exit(void)
{
    cdev_del(&qcom_dev->cdev);
    kfree(qcom_dev);
    unregister_chrdev_region(devno, 1);
    pr_info("qcom_char: unloaded\n");
}

module_init(qcom_init);
module_exit(qcom_exit);
MODULE_LICENSE("GPL");
MODULE_DESCRIPTION("Qualcomm Character Device Example");
```

### 用户态测试程序

```c
// test.c — 编译：gcc test.c -o test
#include <stdio.h>
#include <stdlib.h>
#include <fcntl.h>
#include <unistd.h>
#include <sys/ioctl.h>

int main()
{
    int fd = open("/dev/qcom_char", O_RDWR);
    if (fd < 0) {
        perror("open");
        return 1;
    }
    
    // 写入数据
    write(fd, "Hello from user!", 16);
    
    // 读取数据
    char buf[64] = {0};
    read(fd, buf, sizeof(buf));
    printf("Read: %s\n", buf);
    
    // 发送 ioctl 命令
    ioctl(fd, 0);  // 清空缓冲区
    
    close(fd);
    return 0;
}
```

## 自动创建设备节点 — udev

传统方式需要手动 `mknod /dev/qcom_char c 241 0`，现代 Linux 使用 **udev** 自动管理：

```bash
# 在驱动中创建 class 和 device
struct class *qcom_class;
struct device *qcom_device;

// 在 init 中添加：
qcom_class = class_create(THIS_MODULE, "qcom");
qcom_device = device_create(qcom_class, NULL, devno, NULL, "qcom_char");

// 在 exit 中清理：
device_destroy(qcom_class, devno);
class_destroy(qcom_class);
```

## 关键知识点

### 1. 用户态 vs 内核态数据拷贝

```c
// 错误：直接解引用用户态指针（会导致内核崩溃）
// copy_from_user / copy_to_user 是必须的
```

### 2. unlocked_ioctl vs 旧的 ioctl

- 内核 2.6.36+ 使用 `unlocked_ioctl`（不再使用大内核锁）
- 高通代码中常见 `unlocked_ioctl` 配合 `_IOR/_IOW/_IO` 宏

## 面试考点

1. **open 调用流程**：VFS → 根据设备号找到 cdev → 调用 fops->open
2. **container_of 宏**：从结构体成员反推出结构体地址
3. **copy_from_user 失败会怎样**：返回非 0 值，驱动应该返回 -EFAULT
4. **为什么需要 filp->private_data**：用于在多个 open 实例间区分设备状态
5. **udev 工作机制**：内核通过 uevent 通知用户态，udev 在 /dev 下创建设备节点
